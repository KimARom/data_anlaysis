result_lst = []
threshold_lst = []

val_y_prob = grid.predict_proba(X_valid)[:, 1] # 1(생존)일 확률

for n in range(0, 100):
    threshold = round(((n + 1) * 0.01), 2)
    threshold_lst.append(threshold)
    pred_y = val_y_prob.copy()
    pred_y = np.where(pred_y > threshold, 1., 0.)

    result_lst.append(accuracy_score(y_valid, pred_y))

idx = np.argmax(result_lst)
